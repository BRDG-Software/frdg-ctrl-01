PI MQTT Local Broker Setup
// needed for led control
https://cedalo.com/blog/mqtt-broker-raspberry-pi-installation-guide/

- sudo apt update && sudo apt upgrade

- sudo apt install -y mosquitto mosquitto-clients
	//start mosquitto with
- mosquitto -v
	//restart mosquitto
- sudo systemctl restart mosquitto
	//change the port (from default 1883) if conflictz
- cd /etc/mosquitto
	//create new local config
- sudo nano /conf.d/connect.conf
- inside connect.conf
	- listener 1883
	- allow_anonymous true
	//run mosquitto using the new conf file
- mosquitto -c /etc/mosquitto/conf.d/connect.conf

//\\ run this on the pi using a /home/[USER]/.config/autostart/FILNAME.desktop

	//include the local broker path and password(?) in .env file
- NEXT_PUBLIC_DB_HOST="LOCALIP"
- MQTT_BROKER="mqtt://LOCALIP:1883"
- MQTT_PASSWORD="public"

	//check if the mqtt broker is running
- sudo ss -tulwn
	//should show that port 1883 is listening

	//subscribe to mqtt topic to see output in terminal
- mosquitto_sub -h BROKERIP -p PORT -t FRDG/SHLV/leds
