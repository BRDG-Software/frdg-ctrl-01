PI MQTT Local Broker Setup
https://cedalo.com/blog/mqtt-broker-raspberry-pi-installation-guide/

- sudo apt update && sudo apt upgrade

- sudo apt install -y mosquitto mosquitto-clients
	//start mosquitto with
- mosquitto -v
	//restart mosquitto
- sudo systemctl restart mosquitto
	//change the port (from default 1883) if conflictz
- cd /etc/mosquitto
- sudo nano mosquitto.conf
- add "listener 1887 0.0.0.0" to listen to all on port 1887
	//verify port has changed
- sudo netstat -tulnp | grep mosquitto
